# Spinning turret bot

5 SETSPEED					# Constant speed
0 SETTRACKS					# Go straight up (we'll hit the wall though)

main_loop:
	AIM 10 + SETAIM			# Rotate turret
	SCAN 0 > "fire" CALLIF	# Call fire if found
	"main_loop" JUMP  		# Otherwise keep scanning

fire:
	8 FIRE					# Fire with power 8
	RETURN					# Back to scanning
